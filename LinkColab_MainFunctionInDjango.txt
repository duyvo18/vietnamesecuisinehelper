def handle_uploaded_file(f):
    filename = datetime.now().strftime('%Y%m%d%H%M%S')
    with open('files/%s.jpg'%(filename), 'wb+') as destination:
        for chunk in f.chunks():
            destination.write(chunk)
        return filename
